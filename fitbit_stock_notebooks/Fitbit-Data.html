def intervalize(dataframe, mean, std, col):
    one_std         = mean + std
    two_std         = mean + (2*std)
    neg_one_std     = mean - std
    neg_two_std     = mean - (2*std)
    if ((dataframe[col] < one_std) and (dataframe[col] > neg_one_std)):
        return 0
    elif dataframe[col] > two_std:
        return 2
    elif dataframe[col] < neg_two_std:
        return -2
    elif dataframe[col] > one_std:
        return 1
    elif dataframe[col] < neg_one_std:
        return -1

def chunk_std(dataframe, column):
    dataframe[column+'_delta'] = abs(dataframe[column].diff().shift(-1))
    cols = ['span','min_per','max_per','start_min','min_heart','max_heart']
    std  = dataframe[column+'_delta'].std()
    new_df = pd.DataFrame()
    new_chunk = pd.DataFrame()
    start_chunk = None
    min_per, max_per      = dataframe.iloc[0][column],dataframe.iloc[0][column]
    start_min, span       = dataframe.iloc[0].minutes_time,1
    min_heart, max_heart  = dataframe.iloc[0].heart, dataframe.iloc[0].heart
    for curr in dataframe.index:
        span+=1
        #print(dataframe.loc[curr][column+'_delta'])
        if dataframe.loc[curr][column+'_delta'] > std:
            #TODO: ADD TOGGLE TO SWITCH TO ACCELERATED GROWTH AND NO GROWTH
            start_min=dataframe.loc[curr].minutes_time
            start_chunk = curr
            new_chunk = pd.DataFrame(data = [[span, min_per, max_per, start_min, min_heart, max_heart]], index =[start_chunk], columns=list(cols))
            new_df = new_df.append(new_chunk)
            min_per, max_per      = dataframe.loc[curr][column], dataframe.loc[curr][column]
            min_heart, max_heart  = dataframe.loc[curr]['heart'],   dataframe.loc[curr]['heart']
            span = 1
            #print("Higher Creating new chunk",new_df)
        else:
            if (min_per > dataframe.loc[curr][column]):
                min_per = dataframe.loc[curr][column]
            if (max_per < dataframe.loc[curr][column]):
                max_per = dataframe.loc[curr][column]
            if (min_heart > dataframe.loc[curr]['heart']):
                min_heart = dataframe.loc[curr]['heart']
            if (max_heart < dataframe.loc[curr]['heart']):
                max_heart = dataframe.loc[curr]['heart']
    return new_df
